async function callClaude(prompt, apiKey) {

  const makeReq = (messages) => fetch('https://api.anthropic.com/v1/messages', {

    method: 'POST',

    headers: {

      'Content-Type': 'application/json',

      'x-api-key': apiKey,

      'anthropic-version': '2023-06-01',

      'anthropic-dangerous-direct-browser-access': 'true'

    },

    body: JSON.stringify({

      model: 'claude-haiku-4-5-20251001',

      max_tokens: 2000,

      tools: [{ type: 'web_search_20250305', name: 'web_search' }],

      messages

    })

  });


  let messages = [{ role: 'user', content: prompt }];


  // Agentic loop - follow tool_use until we get end_turn with text

  for (let i = 0; i < 5; i++) {

    const resp = await makeReq(messages);

    const data = await resp.json();

    const blockTypes = (data.content||[]).map(b=>b.type).join(', ');

    log(`Turn ${i+1}: HTTP ${resp.status} | stop: ${data.stop_reason} | blocks: ${blockTypes}`, '');


    if (!resp.ok) {

      throw new Error(`API ${resp.status}: ${data.error?.message || JSON.stringify(data).slice(0,200)}`);

    }


    // Collect any text blocks

    const textBlocks = (data.content||[]).filter(b => b.type === 'text');

    if (textBlocks.length > 0) {

      return textBlocks.map(b => b.text).join('\n');

    }


    if (data.stop_reason === 'tool_use') {

      // Add assistant message, then tool results

      messages.push({ role: 'assistant', content: data.content });

      const toolResults = (data.content||[])

        .filter(b => b.type === 'tool_use')

        .map(tb => ({ type: 'tool_result', tool_use_id: tb.id, content: '' }));

      messages.push({ role: 'user', content: toolResults });

      continue;

    }


    throw new Error(`Unexpected stop_reason: ${data.stop_reason}. Blocks: ${blockTypes}`);

  }


  throw new Error('Max turns reached without text response');

}


async function fetchMeetData(url, apiKey) {

  const target = document.getElementById('targetEvent').value.trim();

  const prompt = `Search the web for the current status of this track & field meet: ${url}


Visit the page and find the full event schedule. Look for:

1. A list of all events with their order/numbers

2. Which event has a "Live" indicator (currently happening)

3. Which events are completed vs upcoming


Then respond with ONLY a JSON code block (no explanation), like this:

\`\`\`json

{

  "events": [

    { "number": 1, "name": "Boys 100m Dash", "status": "done" },

    { "number": 2, "name": "Girls 100m Dash", "status": "live" },

    { "number": 3, "name": "Boys 200m Dash", "status": "upcoming" }

  ],

  "liveEvent": "Girls 100m Dash",

  "liveIndex": 1

}

\`\`\`


Status must be exactly: "done", "live", or "upcoming".

liveIndex is the 0-based array index of the live event.

The target event I care about is: "${target}" — make sure it appears in the list if it's on the schedule.

If you cannot load the page or find events, return: \`\`\`json\n{"events":[],"liveEvent":null,"liveIndex":-1}\`\`\``;


  const raw = await callClaude(prompt, apiKey);

  log(`Raw response (first 200 chars): ${raw.slice(0,200)}`, '');


  // Try extracting JSON from code block first, then bare JSON

  let jsonStr = null;

  const codeBlock = raw.match(/```json\s*([\s\S]*?)```/);

  if (codeBlock) {

    jsonStr = codeBlock[1].trim();

  } else {

    // Fall back to finding outermost { ... }

    const bare = raw.match(/\{[\s\S]*\}/);

    if (bare) jsonStr = bare[0];

  }


  if (!jsonStr) throw new Error(`Could not find JSON in response. Raw: ${raw.slice(0,300)}`);


  try {

    return JSON.parse(jsonStr);

  } catch(e) {

    throw new Error(`JSON parse failed: ${e.message}. Raw JSON: ${jsonStr.slice(0,200)}`);

  }

}


function updateUI(data) {

  const target = document.getElementById('targetEvent').value.trim().toLowerCase();

  const alertOffset = parseInt(document.getElementById('alertOffset').value) || 3;


  state.events = data.events || [];

  state.currentLiveIndex = data.liveIndex ?? -1;

  state.targetIndex = state.events.findIndex(e => e.name.toLowerCase().includes(target));


  // Track timestamps for avg estimation

  if (state.currentLiveIndex >= 0) {

    const now = Date.now();

    const existing = state.eventTimestamps.find(t => t.index === state.currentLiveIndex);

    if (!existing) {

      state.eventTimestamps.push({ index: state.currentLiveIndex, time: now });

      // Compute avg from last few transitions

      if (state.eventTimestamps.length >= 2) {

        const sorted = [...state.eventTimestamps].sort((a,b) => a.index - b.index);

        const diffs = [];

        for (let i = 1; i < sorted.length; i++) {

          const dt = (sorted[i].time - sorted[i-1].time) / 60000;

          const di = sorted[i].index - sorted[i-1].index;

          if (di > 0) diffs.push(dt / di);

        }

        if (diffs.length) state.avgEventDuration = diffs.reduce((a,b)=>a+b,0)/diffs.length;

      }

    }

  }


  // Update current event display

  const liveEvent = state.events[state.currentLiveIndex];

  const display = document.getElementById('currentEventDisplay');

  document.getElementById('liveDotLabel').style.display = liveEvent ? 'flex' : 'none';


  if (liveEvent) {

    display.innerHTML = `

      <div class="event-number">#${state.currentLiveIndex + 1}</div>

      <div>

        <div class="event-name">${liveEvent.name}</div>

        <div class="event-meta">Event ${liveEvent.number ?? state.currentLiveIndex + 1} of ${state.events.length}</div>

      </div>`;

  } else {

    display.innerHTML = `<div class="empty-state">Could not identify live event.<br>Will retry on next check.</div>`;

  }


  // ETA

  const etaRow = document.getElementById('etaRow');

  if (state.targetIndex >= 0 && state.currentLiveIndex >= 0) {

    const eventsLeft = state.targetIndex - state.currentLiveIndex;

    document.getElementById('etaEvents').textContent = eventsLeft > 0 ? `${eventsLeft} event${eventsLeft !== 1 ? 's' : ''}` : eventsLeft === 0 ? 'NOW' : 'Passed';

    if (state.avgEventDuration && eventsLeft > 0) {

      const mins = Math.round(state.avgEventDuration * eventsLeft);

      document.getElementById('etaTime').textContent = `~${mins} min`;

      document.getElementById('etaAvg').textContent = `~${Math.round(state.avgEventDuration)} min/event`;

      const startTime = new Date(Date.now() + mins * 60000);

      document.getElementById('etaStart').textContent = startTime.toLocaleTimeString([], {hour:'2-digit',minute:'2-digit'});

    } else {

      document.getElementById('etaTime').textContent = 'Estimating…';

      document.getElementById('etaAvg').textContent = 'Estimating…';

      document.getElementById('etaStart').textContent = '—';

    }

    etaRow.style.display = 'flex';

  } else {

    etaRow.style.display = 'none';

  }


  // Events list

  const list = document.getElementById('eventsList');

  if (state.events.length === 0) {

    list.innerHTML = '<div class="empty-state">No events found. The meet may not have started or the page couldn\'t be read.</div>';

  } else {

    document.getElementById('eventsCountLabel').textContent = `${state.events.length} events`;

    list.innerHTML = state.events.map((ev, i) => {

      const isLive = i === state.currentLiveIndex;

      const isTarget = i === state.targetIndex;

      const isDone = ev.status === 'done' && !isLive;

      const eventsFromLive = i - state.currentLiveIndex;

      const isSoon = !isLive && !isTarget && !isDone && eventsFromLive > 0 && eventsFromLive <= alertOffset;


      let cls = '';

      if (isLive) cls = 'is-live';

      else if (isTarget) cls = 'is-target';

      else if (isSoon) cls = 'is-warn';

      else if (isDone) cls = 'is-done';


      let badge = '';

      if (isLive) badge = '<span class="ev-badge badge-live">Live</span>';

      else if (isTarget) badge = '<span class="ev-badge badge-target">Target</span>';

      else if (isSoon) badge = '<span class="ev-badge badge-soon">Soon</span>';

      else if (isDone) badge = '<span class="ev-badge badge-done">Done</span>';


      let eta = '';

      if (!isDone && !isLive && state.currentLiveIndex >= 0 && eventsFromLive > 0 && state.avgEventDuration) {

        const m = Math.round(state.avgEventDuration * eventsFromLive);

        eta = `~${m}m`;

      }


      return `<div class="event-row ${cls}">

        <span class="ev-num">${ev.number ?? i+1}</span>

        ${badge}

        <span class="ev-name">${ev.name}</span>

        <span class="ev-est">${eta}</span>

      </div>`;

    }).join('');


    // Scroll live event into view

    setTimeout(() => {

      const liveRow = list.querySelectorAll('.event-row')[state.currentLiveIndex];

      if (liveRow) liveRow.scrollIntoView({ behavior: 'smooth', block: 'center' });

    }, 100);

  }


  // Check alert condition

  if (!state.alertFired && state.targetIndex >= 0 && state.currentLiveIndex >= 0) {

    const eventsLeft = state.targetIndex - state.currentLiveIndex;

    if (eventsLeft <= alertOffset && eventsLeft > 0) {

      showAlert(`⚡ ${alertOffset} events until ${document.getElementById('targetEvent').value}! Get ready!`);

      state.alertFired = true;

    } else if (eventsLeft <= 0) {

      showAlert(`🏃 ${document.getElementById('targetEvent').value} is happening NOW or next!`);

      state.alertFired = true;

    }

  }

}


let nextCheckCountdown = 0;

let countdownTick = null;


function startCountdown(seconds) {

  clearInterval(countdownTick);

  nextCheckCountdown = seconds;

  const label = document.getElementById('nextCheckLabel');

  const fill = document.getElementById('nextCheckFill');


  function tick() {

    nextCheckCountdown--;

    const pct = ((seconds - nextCheckCountdown) / seconds) * 100;

    fill.style.width = pct + '%';

    if (nextCheckCountdown <= 0) {

      label.textContent = 'Checking now…';

      clearInterval(countdownTick);

    } else {

      label.textContent = `Next check in ${nextCheckCountdown}s`;

    }

  }

  label.textContent = `Next check in ${seconds}s`;

  fill.style.width = '0%';

  countdownTick = setInterval(tick, 1000);

}


async function doCheck() {

  if (isChecking) return;

  isChecking = true;


  const apiKey = document.getElementById('apiKey').value.trim();

  const url = document.getElementById('meetUrl').value.trim();


  document.getElementById('btnCheck').innerHTML = '↻ Checking<span class="spinner"></span>';


  try {

    log(`Starting check... API key length: ${apiKey.length}`, 'info');


    // Step 1: Raw fetch test

    log('Step 1: Sending API request...', 'info');

    let rawResp, rawData;

    try {

      rawResp = await fetch('https://api.anthropic.com/v1/messages', {

        method: 'POST',

        headers: {

          'Content-Type': 'application/json',

          'x-api-key': apiKey,

          'anthropic-version': '2023-06-01',

          'anthropic-dangerous-direct-browser-access': 'true'

        },

        body: JSON.stringify({

          model: 'claude-haiku-4-5-20251001',

          max_tokens: 100,

          messages: [{ role: 'user', content: 'Say the word READY and nothing else.' }]

        })

      });

      log(`Step 1 response: HTTP ${rawResp.status}`, rawResp.ok ? 'success' : 'error');

      rawData = await rawResp.json();

      log(`Step 1 body: ${JSON.stringify(rawData).slice(0,300)}`, rawResp.ok ? '' : 'error');

    } catch(e) {

      log(`Step 1 FAILED (network/CORS?): ${e.message}`, 'error');

      return;

    }


    if (!rawResp.ok) {

      log(`API rejected request. Not proceeding.`, 'error');

      return;

    }


    log('Step 1 passed! Now fetching meet data...', 'success');

    const data = await fetchMeetData(url, apiKey);

    state.lastCheckTime = new Date();


    if (data.events.length > 0) {

      log(`Got ${data.events.length} events. Live: ${data.liveEvent || 'unknown'}`, 'success');

    } else {

      log('No events returned. Page may be loading slowly.', 'warn');

    }


    updateUI(data);

  } catch (e) {

    log(`Error: ${e.message}`, 'error');

    log(`Stack: ${e.stack ? e.stack.slice(0,200) : 'none'}`, 'error');

  } finally {

    isChecking = false;

    document.getElementById('btnCheck').textContent = '↻ Check Now';

  }

}


async function checkNow() {

  await doCheck();

  // Restart countdown

  if (monitorInterval) {

    clearInterval(monitorInterval);

    const mins = parseFloat(document.getElementById('checkInterval').value) || 2;

    startCountdown(mins * 60);

    monitorInterval = setInterval(async () => {

      await doCheck();

      startCountdown(mins * 60);

    }, mins * 60 * 1000);

  }

}


async function startMonitoring() {

  const apiKey = document.getElementById('apiKey').value.trim();

  if (!apiKey || !apiKey.startsWith('sk-ant')) {

    alert('Please enter a valid Anthropic API key (starts with sk-ant-)');

    return;

  }


  // Request notification permission

  if ('Notification' in window && Notification.permission === 'default') {

    await Notification.requestPermission();

  }


  state.alertFired = false;

  state.eventTimestamps = [];


  document.getElementById('btnStart').disabled = true;

  document.getElementById('btnStop').disabled = false;

  document.getElementById('btnCheck').disabled = false;

  setStatus('monitoring');

  log('Monitoring started.', 'info');


  const mins = parseFloat(document.getElementById('checkInterval').value) || 2;

  await doCheck();

  startCountdown(mins * 60);


  monitorInterval = setInterval(async () => {

    await doCheck();

    startCountdown(mins * 60);

  }, mins * 60 * 1000);

}


function stopMonitoring() {

  clearInterval(monitorInterval);

  clearInterval(countdownTick);

  monitorInterval = null;

  document.getElementById('btnStart').disabled = false;

  document.getElementById('btnStop').disabled = true;

  document.getElementById('btnCheck').disabled = true;

  document.getElementById('nextCheckLabel').textContent = '';

  document.getElementById('nextCheckFill').style.width = '0%';

  setStatus('idle');

  log('Monitoring stopped.', 'warn');

}
